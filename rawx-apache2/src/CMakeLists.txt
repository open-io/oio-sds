add_definitions(-DG_LOG_DOMAIN="oio.rawx.httpd")

include_directories(BEFORE
	${CMAKE_CURRENT_SOURCE_DIR}
	${CMAKE_SOURCE_DIR}
	${CMAKE_BINARY_DIR})

		${ATTR_INCLUDE_DIRS}
		${APACHE2_INCLUDE_DIRS}
		${APR_INCLUDE_DIRS}
		${LZO_INCLUDE_DIRS}
		${ZLIB_INCLUDE_DIRS}
		${ZMQ_INCLUDE_DIRS})

		${APACHE2_LIBRARY_DIRS}
		${APR_LIBRARY_DIRS}
		${LZO_LIBRARY_DIRS}
		${ZLIB_LIBRARY_DIRS})

		mod_dav_rawx.c
		rawx_bucket.c

		${ZMQ_LIBRARIES})
